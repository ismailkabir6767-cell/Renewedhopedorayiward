<script>
function save(){
 let n=name.value, nin2=nin.value, p=phone.value, d=dob.value;
 if(!n||!nin2||!p||!d){alert("Cika duk bayanai");return;}

 let data=JSON.parse(localStorage.getItem("reg"))||[];
 data.push({
  Name:n,
  NIN:nin2,
  Phone:p,
  DOB:d,
  Time:new Date().toLocaleString()
 });
 localStorage.setItem("reg",JSON.stringify(data));
 location="success.html";
}
</script>
