<script>
function saveData() {

let name = document.getElementById("name").value;
let nin = document.getElementById("nin").value;
let phone = document.getElementById("phone").value;
let dob = document.getElementById("dob").value;

if(name=="" || nin=="" || phone=="" || dob==""){
    alert("Cika dukkan bayanai");
    return;
}

let records = JSON.parse(localStorage.getItem("registrations")) || [];

records.push({
    FullName: name,
    NIN: nin,
    Phone: phone,
    DOB: dob
});

localStorage.setItem("registrations", JSON.stringify(records));

document.getElementById("msg").style.display = "block";

document.getElementById("name").value="";
document.getElementById("nin").value="";
document.getElementById("phone").value="";
document.getElementById("dob").value="";
}
</script>
