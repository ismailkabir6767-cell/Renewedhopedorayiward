<script>
function save(){
 let n=document.getElementById("name").value;
 let nin=document.getElementById("nin").value;
 let p=document.getElementById("phone").value;
 let d=document.getElementById("dob").value;

 if(n==""||nin==""||p==""||d==""){alert("Cika duk bayanai");return;}

 let data=JSON.parse(localStorage.getItem("reg"))||[];
 data.push({Name:n,NIN:nin,Phone:p,DOB:d});
 localStorage.setItem("reg",JSON.stringify(data));

 window.location="success.html";
}
